Y.grep(arr, callback)
过滤出符合条件的数组 

var arr =[-3,-2,0,2,4];
Y.grep(arr,function(item,i){
    return  item > 0;
});
console.log(arr); //[2,4]

Y.map(arr, callback)
对数组的每项进行处理,返回新数组 

var arr =[1,2,3];
Y.map(arr,function(item,i){
    return item * 3;
});

Y.loadScript(url,callback)
加载js或者css文件,执行回调函数.
具有缓存(如存在即使用该js)和队列特性(js加载完成后队列触发多个回调). 

Y.loadScript("./lib/test.js",function(){
    dosomething();
});
Y.loadScript("./lib/test.js",function(){
    dosomething2();
});
//test.js加载后依次执行dosomething();dosomething2();

Y.ajax({options}) 

参数说明:
options: {
    url:"请求url地址", {String}
    data:"请求参数",   {Object}
    method:"请求post或者get方法", {"get" | "post"} 默认:"get"
    dataType:"返回的数据类型", {"xml" | "json" | "jsonp"} 默认:"json"
    success:"成功回调函数", {Function}
    error:"失败回调函数",   {Function}
}
eg:
    Y.ajax({
        url:"server.php",
        data:{id:12,name:"human"},
        success:function(ret){
            console.log("response data:" + ret);
        },
        error:function(data){
            console.log("request error");
        },
    });

Y.get(url,data,callback) 

eg:
    Y.get("server.php",{id:22,name:"human"},function(data){
         console.log("res data:" + data);
    });

    Y.get("server.php",function(data){
         console.log("res data:" + data);
    });
